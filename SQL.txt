-- Users
CREATE TABLE users (
    user_id UUID PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    phone VARCHAR(20) UNIQUE,
    full_name VARCHAR(255),
    kyc_status ENUM('pending', 'verified', 'rejected') DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Business Profiles
CREATE TABLE business_profiles (
    profile_id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(user_id),
    business_name VARCHAR(255),
    sector ENUM('retail', 'restaurant', 'ecommerce', 'services', 'manufacturing', 'construction'),
    establishment_date DATE,
    employees_count INT,
    revenue_model VARCHAR(100),
    registration_no VARCHAR(100),
    created_at TIMESTAMP DEFAULT NOW()
);

-- Bank Accounts
CREATE TABLE bank_accounts (
    account_id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(user_id),
    account_number_hash VARCHAR(255) NOT NULL,  -- Hashed
    bank_code VARCHAR(10),
    account_type ENUM('checking', 'savings', 'business'),
    balance DECIMAL(15,2),
    currency CHAR(3) DEFAULT 'SAR',
    opened_at TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Transactions
CREATE TABLE transactions (
    transaction_id UUID PRIMARY KEY,
    account_id UUID REFERENCES bank_accounts(account_id),
    amount DECIMAL(15,2) NOT NULL,
    currency CHAR(3) DEFAULT 'SAR',
    transaction_type ENUM('credit', 'debit'),
    description TEXT,
    category_ml VARCHAR(50),  -- ML predicted
    category_confidence DECIMAL(4,3),
    is_personal BOOLEAN,
    is_business BOOLEAN,
    is_anomaly BOOLEAN DEFAULT FALSE,
    anomaly_score DECIMAL(4,3),
    timestamp TIMESTAMP NOT NULL,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Readiness Assessments
CREATE TABLE readiness_assessments (
    assessment_id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(user_id),
    profile_id UUID REFERENCES business_profiles(profile_id),
    readiness_score INT CHECK (readiness_score BETWEEN 0 AND 100),
    risk_level ENUM('low', 'medium', 'high', 'critical'),
    assessment_version VARCHAR(10),
    status ENUM('active', 'expired', 'superseded'),
    created_at TIMESTAMP DEFAULT NOW(),
    expires_at TIMESTAMP
);

-- Assessment Factors
CREATE TABLE assessment_factors (
    factor_id UUID PRIMARY KEY,
    assessment_id UUID REFERENCES readiness_assessments(assessment_id),
    factor_name VARCHAR(50),
    weight DECIMAL(4,2),
    raw_score DECIMAL(5,2),
    weighted_score DECIMAL(5,2),
    explanation TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Recommendations
CREATE TABLE recommendations (
    recommendation_id UUID PRIMARY KEY,
    assessment_id UUID REFERENCES readiness_assessments(assessment_id),
    priority INT CHECK (priority BETWEEN 1 AND 5),
    category VARCHAR(50),
    action_title VARCHAR(255),
    action_description TEXT,
    expected_impact VARCHAR(100),
    estimated_duration_days INT,
    is_completed BOOLEAN DEFAULT FALSE,
    completed_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Conversations
CREATE TABLE conversations (
    conversation_id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(user_id),
    session_id UUID,
    message_text TEXT,
    intent VARCHAR(50),
    sentiment VARCHAR(20),
    timestamp TIMESTAMP DEFAULT NOW(),
    is_bot BOOLEAN
);

-- Bank Policies
CREATE TABLE bank_policies (
    policy_id UUID PRIMARY KEY,
    policy_name VARCHAR(255),
    min_readiness_score INT,
    max_risk_level ENUM('low', 'medium', 'high'),
    min_monthly_revenue DECIMAL(15,2),
    min_account_age_months INT,
    sector_restrictions JSON,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Audit Logs
CREATE TABLE audit_logs (
    log_id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(user_id),
    action_type VARCHAR(50),
    entity_type VARCHAR(50),
    entity_id UUID,
    old_value JSONB,
    new_value JSONB,
    timestamp TIMESTAMP DEFAULT NOW(),
    ip_address INET
);
